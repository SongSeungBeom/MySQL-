SELECT 
	AVG(review_count)
FROM
	(SELECT
		SUBSTRING(address, 1, 2) AS region,
        COUNT(*) AS review_count
	FROM
		review AS r LEFT OUTER JOIN member AS m
	ON r.mem_id = m.id
    GROUP BY SUBSTRING(address, 1, 2)
    HAVING
		region IS NOT NULL
        AND region != '안드') AS review_count_summary
